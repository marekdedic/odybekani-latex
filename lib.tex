% SONG LIST FORMATTING
%------------------------------------------------------------------------------

% Length registers for calculating chord width accumulation across lines
\newlength{\songchordlen}
\newlength{\songchordtmp}

% \songlistentry{artist}{title} - Formats a single entry in the song list
% Renders as: title ......................... artist
\newcommand{\songlistentry}[2]{%
	#2 \hfill #1 \\
}

% \songlistdivider - Draws a dashed line separator between song entries
% Uses negative vspace to compress the visual gap
\newcommand{\songlistdivider}{%
	\vspace{-20pt} \\
	{\color{darkgray} \hdashrule{1.01\textwidth}{1pt}{1pt 2pt}} \\
}

% Custom multicols environment that gracefully handles single-column layout
% When columns=1, bypasses the multicols environment entirely to avoid overhead
\let\multicolmulticols\multicols
\let\endmulticolmulticols\endmulticols
\RenewDocumentEnvironment{multicols}{mO{}}{%
	\ifnum#1=1
		#2%
	\else
		\multicolmulticols{#1}[#2]
	\fi
}{%
	\ifnum#1=1
	\else
		\endmulticolmulticols
	\fi
}

% Page style for the List of Songs (table of contents)
% Displays version number in footer, centered on odd pages, right on even
\newpagestyle{losstyle}{%
	\footrule
	\setfoot[\songurlfont \songversion]
		[][]{}{}
		{\songurlfont \songversion}
}

% \listofsongs[columns] - Generates the index/table of contents page
% Optional argument: number of columns (default: 1)
% Uses Lua script to generate formatted list of all songs
\newcommand{\listofsongs}[1][1]{%
	\clearpage
	\newgeometry{a5paper,twoside,top=50pt,bottom=50pt,inner=70pt,outer=55pt,headheight=25pt,headsep=25pt,footskip=20pt}
	\begin{multicols}{#1}
		\directlua{print_los("\jobname")}
	\end{multicols}
	\thispagestyle{losstyle}
	\restoregeometry
}

%------------------------------------------------------------------------------
% SONG ENVIRONMENT
%------------------------------------------------------------------------------

% Storage box for measuring rendered song content height
\newsavebox{\songbox}

% \songmeasurebox - Calculates total height of songbox (height + depth)
\newcommand\songmeasurebox{\dimexpr\ht\songbox+\dp\songbox\relax}

% \begin{song}[author]{number}{title}{url} ... \end{song}
% Main environment for typesetting a single song
% Arguments: #1 = author, #2 = number, #3 = title, #4 = url
%
% At begin: Initializes Lua recording with song metadata
% At end: Captures rendered content, handles page breaks, applies song style
\newenvironment{song}[4][]{%
	\directlua{startrecording("#2", "#3", "#1", "#4", "\jobname")}
}{%
	\directlua{stoprecording()}
	\begin{lrbox}{\songbox}\noindent\begin{minipage}{\linewidth}\directlua{print_song()}\end{minipage}\end{lrbox}
	\clearpage
	\ifodd\value{page}
		\ifthenelse{\songmeasurebox>\textheight}{\null\clearpage}{}
	\fi
	\thispagestyle{songstyle}
	\directlua{print_song()}
	\renewcommand{\songauthormark}{}
	\preshortmark\songauthormark
}

%------------------------------------------------------------------------------
% SONG FORMATTING COMMANDS
%------------------------------------------------------------------------------

% \songchordbox{chord} - Creates box for chord symbols above text line
% Places chord with spacing to avoid overlapping lyrics
\newcommand{\songchordbox}[1]{\makebox[\width][l]{\rule{0pt}{2.2em}\raisebox{1em}{\textbf{#1}}}\kern0.2em}

\newcommand{\songpart}[1]{%
	\begin{samepage}#1\end{samepage}
}

\newcommand{\songpartmarker}[1]{%
	\noindent%
	\makebox[0pt][r]{%
		\raisebox{-0.15em}{%
			\textcolor{songpartmarkercolor}{%
				\songpartmarkerfont \huge \textbf{#1}%
			}
		}
		\hspace*{0.5em}
	}%
	\ignorespaces
}

% \songchord{chord} - Renders a chord above the current text position
% Accumulates chord width for proper alignment across the line
\newcommand{\songchord}[1]{\settowidth{\songchordtmp}{\songchordbox{#1}}\addtolength{\songchordlen}{\songchordtmp}\songchordbox{#1}}

% \songverse{number} - Typesets verse marker (e.g., "1.", "2.")
\newcommand{\songverse}[1]{\songpartmarker{#1.}\kern-0.05em}

% \songchorus - Typesets "R:" refrain/chorus marker in margin
\newcommand{\songchorus}{\songpartmarker{R:}}

% \songchordkern - Resets chord length counter after line break
% Prevents chord width accumulation across line breaks
\newcommand{\songchordkern}{\kern-\songchordlen\setlength{\songchordlen}{0em}}

% \songrepeatstart, \songrepeatend - Repeat markers "[:" and ":]"
\newcommand{\songrepeatstart}{\textbf{[:} }
\newcommand{\songrepeatend}{ \textbf{:]}}

%------------------------------------------------------------------------------
% SONG MARKS & PAGE STYLE
%------------------------------------------------------------------------------

% Song metadata marks - hold current song's name, author, URL
% Populated by \songsettitleurl and \songsetauthor, used in headers/footers
\newcommand{\songnamemark}{}
\newcommand{\songauthormark}{}
\newcommand{\songurlmark}{}

% Create short marks for running headers (condensed versions for page top)
\newshortmark\songnamemark
\newshortmark\songauthormark
\newshortmark\songurlmark

% Song page style - applied to all song pages
% Header: song title (centered on odd, right on even), author below
% Footer: URL on right side
\newpagestyle{songstyle}{%
	\footrule
	\sethead[]
		[
			\parbox{\textwidth}{%
				\centering \songnamefont \topshortmark\songnamemark \\[-2pt]
				\rule{\textwidth}{0.6pt} \\
				\raggedleft \songauthorfont \topshortmark\songauthormark \hspace*{3em}
			}
		]
		[]
		{}
		{%
			\parbox{\textwidth}{%
				\centering \songnamefont \topshortmark\songnamemark \\[-2pt]
				\rule{\textwidth}{0.6pt} \\
				\raggedright \songauthorfont \hspace*{3em} \topshortmark\songauthormark
			}
		}
		{}

	\setfoot[\songurlfont \topshortmark\songurlmark]
		[][]{}{}
		{\songurlfont \topshortmark\songurlmark}
}

%------------------------------------------------------------------------------
% SONG METADATA COMMANDS
%------------------------------------------------------------------------------

% \songsettitleurl{title}{url} - Sets song title and URL for headers
% Arguments: #1 = song title, #2 = URL (stored but currently unused)
\newcommand{\songsettitleurl}[2]{%
	\renewcommand{\songnamemark}{#1}
	\preshortmark\songnamemark
	%\renewcommand{\songurlmark}{\href{https://#2}{#2}}
	\preshortmark\songurlmark
}

% \songsetauthor{author} - Sets song author/composer for headers
% Also adds a line break after author in the song body
\newcommand{\songsetauthor}[1]{%
	\renewcommand{\songauthormark}{#1}
	\preshortmark\songauthormark
	\leavevmode \\
}

% TITLE PAGE & HEADER IMAGES
%------------------------------------------------------------------------------

% \songtitlepage - Generates the songbook's title page
% Uses template-defined fonts and logo, centers content vertically
% Enables header images on all subsequent pages after title page
\newcommand{\songtitlepage}{%
	\begin{titlepage}
		\centering
		\null
		\vspace{0.3cm}
		{\songtitletopfont \uppercase{\songtitletop}}

		\vfill
		\includegraphics[width=0.75\textwidth]{\titlepagelogo}
		\vfill

		{\songtitlebottomfont \uppercase{\songtitlebottom}}
		\vspace{1.5cm}
		\null
	\end{titlepage}
	\addheaderimg
}

% Disables paragraph indentation throughout the document
\setlength{\parindent}{0pt}

% \addheaderimg - Adds the between-page logo to all subsequent pages
% Places logo in outer margin (left on odd pages, right on even)
% Uses eso-pic to add background image to each page after shipout
\newcommand{\addheaderimg}{%
	\AddToShipoutPictureBG{%
		\put(-60,0){%
			\parbox[b][\paperheight + 20pt]{\paperwidth + 120pt}{%
				\ifodd\value{page}
				\else
					\raggedleft
				\fi
				\includegraphics[width=120pt]{\betweenpagelogo}
				\vfill
			}
		}
	}
}
